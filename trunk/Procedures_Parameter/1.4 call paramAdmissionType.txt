CREATE PROCEDURE paramAdmissionType() 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
	
select ADMISSION_TYPE_CODE,ADMISSION_TYPE_NAME
from
(
SELECT distinct ADMISSION_TYPE_CODE , ADMISSION_TYPE_NAME
FROM DIM_ADMISSION_TYPE
where ADMISSION_TYPE_CODE not in ('N/A')

union
SELECT 'All' as ADMISSION_TYPE_CODE
,'All' as ADMISSION_TYPE_NAME
from sysibm.sysdummy1
)ADMISSION_TYPE
order by ADMISSION_TYPE_CODE
;
	
    OPEN result_set_1;
END
