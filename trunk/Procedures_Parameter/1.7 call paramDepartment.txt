CREATE PROCEDURE paramDepartment() 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
	
select DEPARTMENT_CODE
,DEPARTMENT
from
(
SELECT distinct DEPARTMENT_CODE
, DEPARTMENT_CODE||'-'||DEPARTMENT_NAME as DEPARTMENT
FROM DIM_FIELD
where DEPARTMENT_CODE not in ('N/A')
union 
SELECT 'All' as DEPARTMENT_CODE
, 'All' as DEPARTMENT
FROM DIM_FIELD
)FIELD
order by DEPARTMENT_CODE

	
    OPEN result_set_1;
END
