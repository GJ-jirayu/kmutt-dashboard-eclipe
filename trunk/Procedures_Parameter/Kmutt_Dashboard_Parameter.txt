-- Param_ACADEMIC_YEAR (ปีการศึกษา)
SELECT distinct ACADEMIC_YEAR
FROM DIM_SEMESTER
where ACADEMIC_YEAR not in ('0')
order by ACADEMIC_YEAR DESC

-- Param_SEMESTER (เทอม)
SELECT distinct SEMESTER_CODE
FROM DIM_SEMESTER
where SEMESTER_CODE not in ('N/A')
order by SEMESTER_CODE ASC

-- Param_EDUCATION_LEVEL (ระดับการศึกษา)
SELECT distinct EDUCATION_LEVEL_CODE , EDUCATION_LEVEL_NAME
FROM DIM_EDUCATION_LEVEL
where EDUCATION_LEVEL_CODE not in ('004','005','006','N/A')
order by EDUCATION_LEVEL_CODE ASC

-- Param_ADMISSION_TYPE (ประเภทการรับเข้า)
SELECT distinct ADMISSION_TYPE_CODE , ADMISSION_TYPE_NAME
FROM DIM_ADMISSION_TYPE
where ADMISSION_TYPE_CODE not in ('N/A')
order by ADMISSION_TYPE_CODE ASC
