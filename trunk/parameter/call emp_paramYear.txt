CREATE PROCEDURE emp_paramYear() 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
		
		SELECT CALENDAR_YEAR
		FROM DIM_DATE
		where CALENDAR_YEAR != '9999'
		group by CALENDAR_YEAR
		order by CALENDAR_YEAR desc
	;
    OPEN result_set_1;
END
