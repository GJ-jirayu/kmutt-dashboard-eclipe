CREATE PROCEDURE emp_paramEdu() 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR

	select EDUCATION_LEVEL_CODE
,EDUCATION_LEVEL_NAME
from
(
SELECT EDUCATION_LEVEL_CODE
, EDUCATION_LEVEL_NAME
, '1' as nos 
FROM HR_DIM_EDUCATION_LEVEL
where EDUCATION_LEVEL_CODE in ('41','96')

union 

SELECT EDUCATION_LEVEL_CODE
, EDUCATION_LEVEL_NAME
, '2' as nos 
FROM HR_DIM_EDUCATION_LEVEL
where EDUCATION_LEVEL_CODE in ('42','44')


union 

SELECT EDUCATION_LEVEL_CODE
, EDUCATION_LEVEL_NAME
, '3' as nos 
FROM HR_DIM_EDUCATION_LEVEL
where EDUCATION_LEVEL_CODE in ('43')

union

SELECT 'All' as EDUCATION_LEVEL_CODE
,'รวม' as EDUCATION_LEVEL_NAME
, '4' as nos 
from sysibm.sysdummy1

)kmutt 
order by nos

	;
		
    OPEN result_set_1;
END
