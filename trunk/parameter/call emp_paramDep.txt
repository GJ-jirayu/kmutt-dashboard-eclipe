CREATE PROCEDURE emp_paramDep() 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
		
select ORGENIZATION_CODE
	,ORGENIZATION_NAME
	from
	(
	select 'All' as ORGENIZATION_CODE
	,'รวม' as ORGENIZATION_NAME
	,'2' as nos
	from sysibm.sysdummy1
	
	union
	
        SELECT ORGENIZATION_NAME_B as ORGENIZATION_CODE
	,ORGENIZATION_NAME_B as ORGENIZATION_NAME
	,'1' as nos
	FROM HR_DIM_DEPARTMENT
	group by ORGENIZATION_NAME_B
	)DEPARTMENT
	order by nos,ORGENIZATION_NAME
	;
		
    OPEN result_set_1;
END
