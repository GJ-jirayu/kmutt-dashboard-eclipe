CREATE PROCEDURE emp_by_for_year(IN paramEmpTypeYear VARCHAR(256),IN paramEmpYear VARCHAR(256),IN paramEmpPos VARCHAR(256),IN paramEmpTypeLine VARCHAR(256),IN paramEmpYearSub VARCHAR(256)) 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
		select CALENDAR_YEAR
		,POSITION_GROUP_TYPE_OFFCIATE_NAME
		,no_of_emp
		from
		(	
			SELECT DIM_DATE.CALENDAR_YEAR
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			,count(distinct HR_FACT_EMPLOYEE.EMP_KEY) as no_of_emp
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM HR_FACT_EMPLOYEE
			left join DIM_DATE on DIM_DATE.DATE_KEY = HR_FACT_EMPLOYEE.MONTH_KEY
			left join HR_DIM_POSITION_GROUP on HR_DIM_POSITION_GROUP.POSITION_GROUP_KEY = HR_FACT_EMPLOYEE.POSITION_GROUP_KEY
			left join HR_DIM_EMP_TYPE on HR_DIM_EMP_TYPE.EMP_TYPE_KEY = HR_FACT_EMPLOYEE.EMP_TYPE_KEY
			left join HR_DIM_EMP_LINE_GROUP on HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_KEY = HR_FACT_EMPLOYEE.EMP_LINE_KEY
			left join HR_DIM_NATIONALITY on HR_DIM_NATIONALITY.NATIONALITY_KEY = HR_FACT_EMPLOYEE.NATIONALITY_KEY
			where (NATIONALITY_CODE != '099')
			-- and (DIM_DATE.CALENDAR_YEAR between ('2556') and ('2556'))
			and (DIM_DATE.CALENDAR_YEAR between (paramEmpYear)-(paramEmpYearSub)+(1) and (paramEmpYear))
			and (HR_DIM_POSITION_GROUP.POSITION_GROUP_CODE in (paramEmpPos) or 'All' in (paramEmpPos))
			and (HR_DIM_EMP_TYPE.EMP_TYPE_GROUP_CODE || HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_CODE in (paramEmpTypeLine) or 'All' in (paramEmpTypeLine))
			group by DIM_DATE.CALENDAR_YEAR, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			
			union
			
			SELECT CALENDAR_YEAR
			,'รวม' as POSITION_GROUP_TYPE_OFFCIATE_NAME
			,sum(no_of_emp) as no_of_emp
			,'รวม' as POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM(   	
			SELECT DIM_DATE.CALENDAR_YEAR
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			,count(distinct HR_FACT_EMPLOYEE.EMP_KEY) as no_of_emp
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM HR_FACT_EMPLOYEE
			left join DIM_DATE on DIM_DATE.DATE_KEY = HR_FACT_EMPLOYEE.MONTH_KEY
			left join HR_DIM_POSITION_GROUP on HR_DIM_POSITION_GROUP.POSITION_GROUP_KEY = HR_FACT_EMPLOYEE.POSITION_GROUP_KEY
			left join HR_DIM_EMP_TYPE on HR_DIM_EMP_TYPE.EMP_TYPE_KEY = HR_FACT_EMPLOYEE.EMP_TYPE_KEY
			left join HR_DIM_EMP_LINE_GROUP on HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_KEY = HR_FACT_EMPLOYEE.EMP_LINE_KEY
			left join HR_DIM_NATIONALITY on HR_DIM_NATIONALITY.NATIONALITY_KEY = HR_FACT_EMPLOYEE.NATIONALITY_KEY
			where (NATIONALITY_CODE != '099')
			-- and (DIM_DATE.CALENDAR_YEAR between ('2556') and ('2556'))
			and (DIM_DATE.CALENDAR_YEAR between (paramEmpYear)-(paramEmpYearSub)+(1) and (paramEmpYear))
			and (HR_DIM_POSITION_GROUP.POSITION_GROUP_CODE in (paramEmpPos) or 'All' in (paramEmpPos))
			and (HR_DIM_EMP_TYPE.EMP_TYPE_GROUP_CODE || HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_CODE in (paramEmpTypeLine) or 'All' in (paramEmpTypeLine))
			group by DIM_DATE.CALENDAR_YEAR, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			)kmutt
			group by CALENDAR_YEAR
		)gj	
		order by POSITION_GROUP_TYPE_OFFCIATE_CODE,CALENDAR_YEAR
	;               
		
	DECLARE result_set_2 CURSOR WITH RETURN TO CLIENT FOR

		select FISCAL_YEAR
		,POSITION_GROUP_TYPE_OFFCIATE_NAME

		(	
			SELECT DIM_DATE.FISCAL_YEAR
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME

			left join HR_DIM_EMP_LINE_GROUP on HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_KEY = HR_FACT_EMPLOYEE.EMP_LINE_KEY
			left join HR_DIM_NATIONALITY on HR_DIM_NATIONALITY.NATIONALITY_KEY = HR_FACT_EMPLOYEE.NATIONALITY_KEY
			where (NATIONALITY_CODE != '099')
			-- and (DIM_DATE.FISCAL_YEAR between ('2556') and ('2556'))
			and (DIM_DATE.FISCAL_YEAR between (paramEmpYear)-(paramEmpYearSub)+(1) and (paramEmpYear))
			and (HR_DIM_POSITION_GROUP.POSITION_GROUP_CODE in (paramEmpPos) or 'All' in (paramEmpPos))
			and (HR_DIM_EMP_TYPE.EMP_TYPE_GROUP_CODE || HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_CODE in (paramEmpTypeLine) or 'All' in (paramEmpTypeLine))
			group by DIM_DATE.FISCAL_YEAR, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			
			union
			
			SELECT FISCAL_YEAR
			,'รวม' as POSITION_GROUP_TYPE_OFFCIATE_NAME
			,sum(no_of_emp) as no_of_emp
			,'รวม' as POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM(   	
			SELECT DIM_DATE.FISCAL_YEAR
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			,count(distinct HR_FACT_EMPLOYEE.EMP_KEY) as no_of_emp
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM HR_FACT_EMPLOYEE
			left join DIM_DATE on DIM_DATE.DATE_KEY = HR_FACT_EMPLOYEE.MONTH_KEY
			left join HR_DIM_POSITION_GROUP on HR_DIM_POSITION_GROUP.POSITION_GROUP_KEY = HR_FACT_EMPLOYEE.POSITION_GROUP_KEY
			left join HR_DIM_EMP_TYPE on HR_DIM_EMP_TYPE.EMP_TYPE_KEY = HR_FACT_EMPLOYEE.EMP_TYPE_KEY
			left join HR_DIM_EMP_LINE_GROUP on HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_KEY = HR_FACT_EMPLOYEE.EMP_LINE_KEY
			left join HR_DIM_NATIONALITY on HR_DIM_NATIONALITY.NATIONALITY_KEY = HR_FACT_EMPLOYEE.NATIONALITY_KEY
			where (NATIONALITY_CODE != '099')
			-- and (DIM_DATE.FISCAL_YEAR between ('2556') and ('2556'))
			and (DIM_DATE.FISCAL_YEAR between (paramEmpYear)-(paramEmpYearSub)+(1) and (paramEmpYear))
			and (HR_DIM_POSITION_GROUP.POSITION_GROUP_CODE in (paramEmpPos) or 'All' in (paramEmpPos))
			and (HR_DIM_EMP_TYPE.EMP_TYPE_GROUP_CODE || HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_CODE in (paramEmpTypeLine) or 'All' in (paramEmpTypeLine))
			group by DIM_DATE.FISCAL_YEAR, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			)kmutt
			group by FISCAL_YEAR
		)gj	
		order by POSITION_GROUP_TYPE_OFFCIATE_CODE,FISCAL_YEAR
	
	;
		
    DECLARE result_set_3 CURSOR WITH RETURN TO CLIENT FOR

		select ACADAMIC_YEAR
		,POSITION_GROUP_TYPE_OFFCIATE_NAME
		,no_of_emp
		from
		(	
			SELECT DIM_DATE.ACADAMIC_YEAR
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			,count(distinct HR_FACT_EMPLOYEE.EMP_KEY) as no_of_emp
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM HR_FACT_EMPLOYEE
			left join DIM_DATE on DIM_DATE.DATE_KEY = HR_FACT_EMPLOYEE.MONTH_KEY
			left join HR_DIM_POSITION_GROUP on HR_DIM_POSITION_GROUP.POSITION_GROUP_KEY = HR_FACT_EMPLOYEE.POSITION_GROUP_KEY
			left join HR_DIM_EMP_TYPE on HR_DIM_EMP_TYPE.EMP_TYPE_KEY = HR_FACT_EMPLOYEE.EMP_TYPE_KEY
			left join HR_DIM_EMP_LINE_GROUP on HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_KEY = HR_FACT_EMPLOYEE.EMP_LINE_KEY
			left join HR_DIM_NATIONALITY on HR_DIM_NATIONALITY.NATIONALITY_KEY = HR_FACT_EMPLOYEE.NATIONALITY_KEY
			where (NATIONALITY_CODE != '099')
			-- and (DIM_DATE.ACADAMIC_YEAR between ('2556') and ('2556'))
			and (DIM_DATE.ACADAMIC_YEAR between (paramEmpYear)-(paramEmpYearSub)+(1) and (paramEmpYear))
			and (HR_DIM_POSITION_GROUP.POSITION_GROUP_CODE in (paramEmpPos) or 'All' in (paramEmpPos))
			and (HR_DIM_EMP_TYPE.EMP_TYPE_GROUP_CODE || HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_CODE in (paramEmpTypeLine) or 'All' in (paramEmpTypeLine))
			group by DIM_DATE.ACADAMIC_YEAR, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			
			union
			
			SELECT ACADAMIC_YEAR
			,'รวม' as POSITION_GROUP_TYPE_OFFCIATE_NAME
			,sum(no_of_emp) as no_of_emp
			,'รวม' as POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM(   	
			SELECT DIM_DATE.ACADAMIC_YEAR
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			,count(distinct HR_FACT_EMPLOYEE.EMP_KEY) as no_of_emp
			,HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE
			FROM HR_FACT_EMPLOYEE
			left join DIM_DATE on DIM_DATE.DATE_KEY = HR_FACT_EMPLOYEE.MONTH_KEY
			left join HR_DIM_POSITION_GROUP on HR_DIM_POSITION_GROUP.POSITION_GROUP_KEY = HR_FACT_EMPLOYEE.POSITION_GROUP_KEY
			left join HR_DIM_EMP_TYPE on HR_DIM_EMP_TYPE.EMP_TYPE_KEY = HR_FACT_EMPLOYEE.EMP_TYPE_KEY
			left join HR_DIM_EMP_LINE_GROUP on HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_KEY = HR_FACT_EMPLOYEE.EMP_LINE_KEY
			left join HR_DIM_NATIONALITY on HR_DIM_NATIONALITY.NATIONALITY_KEY = HR_FACT_EMPLOYEE.NATIONALITY_KEY
			where (NATIONALITY_CODE != '099')
			-- and (DIM_DATE.ACADAMIC_YEAR between ('2556') and ('2556'))
			and (DIM_DATE.ACADAMIC_YEAR between (paramEmpYear)-(paramEmpYearSub)+(1) and (paramEmpYear))
			and (HR_DIM_POSITION_GROUP.POSITION_GROUP_CODE in (paramEmpPos) or 'All' in (paramEmpPos))
			and (HR_DIM_EMP_TYPE.EMP_TYPE_GROUP_CODE || HR_DIM_EMP_LINE_GROUP.EMP_LINE_GROUP_CODE in (paramEmpTypeLine) or 'All' in (paramEmpTypeLine))
			group by DIM_DATE.ACADAMIC_YEAR, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_CODE, HR_DIM_POSITION_GROUP.POSITION_GROUP_TYPE_OFFCIATE_NAME
			)kmutt
			group by ACADAMIC_YEAR
		)gj	
		order by POSITION_GROUP_TYPE_OFFCIATE_CODE,ACADAMIC_YEAR
	
	;        	
		
     IF (paramEmpTypeYear) = 'CALENDAR_YEAR' THEN
        OPEN  result_set_1;
     ELSEIF(paramEmpTypeYear) = 'FISCAL_YEAR' THEN
        OPEN  result_set_2;
     ELSE
        OPEN  result_set_3;
     END IF;
END
