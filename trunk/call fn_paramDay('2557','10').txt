CREATE PROCEDURE fn_paramDay(IN paramYear VARCHAR(256),IN paramMonth VARCHAR(256)) 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
	
SELECT VARCHAR_FORMAT(CALENDAR_DATE, 'YYYY-MM-DD') as DATE
,VARCHAR_FORMAT(CALENDAR_DATE, 'DD-MM-YYYY') as CALENDAR_DATE
FROM DIM_DATE
where CALENDAR_YEAR = paramYear
and CALENDAR_MONTH_NO = paramMonth
order by CALENDAR_DATE

	;
    OPEN result_set_1;
END
