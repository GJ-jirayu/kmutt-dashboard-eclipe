CREATE PROCEDURE fn_paramMonth() 
DYNAMIC RESULT SETS 1 
LANGUAGE SQL 
BEGIN 
    DECLARE result_set_1 CURSOR WITH RETURN TO CLIENT FOR
		
	SELECT cast(FISCAL_MONTH_NO as int) as FISCAL_MONTH_NO
	, CALENDAR_TH_MONTH_NAME
	FROM DIM_DATE
	where DATE_KEY != '1'
	group by FISCAL_MONTH_NO,CALENDAR_TH_MONTH_NAME
	order by FISCAL_MONTH_NO

	;
    OPEN result_set_1;
END
